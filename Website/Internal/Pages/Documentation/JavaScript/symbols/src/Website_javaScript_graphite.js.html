 29 /**
 30  * @namespace Event manager
 31  * @class
 32  * @since 1.0 - 2010-02-23
 33  * @version 1.0 - 2010-02-23
 34  */
 35 graphite.events = (function() {
 36   /* Start public */
 37   return {
 38     /**
 39      * Connects a function to an event of a single element
 40      *
 41      * @since 1.0 - 2011-11-03
 42      * @version 1.0 - 2011-11-03
 43      * @param {Object} element The element on which the event is placed.
 44      * @param {Function} eventFunction The function that has to be linked to
 45      *    the event.
 46      * @param {String} eventType Name of the event.
 47      * @param {Boolean} legacy If this value is false or does not exist,
 48      *    the event will be created with "addEventHandler".
 49      * @example
 50         graphite.events.addEvent(
 51           document.getElementById('elementId'),
 52           function() {alert('foo')},
 53           "click",
 54           true
 55         );
 56      */
 57     addEvent: function(element, eventFunction, eventType, legacy) {
 58       //var error;
 59       //error = Estate.Check.VariableType( element, "object" )
 60       //if (error != "") throw new Error( error );
 61       //error = Estate.Check.VariableType( eventFunction, "function" )
 62       //if (error != "") throw new Error( error );
 63 
 64       if (legacy && legacy == true) {
 65         if (element instanceof Array) {
 66           for (var i = 0; i < element.length; i++) {
 67             addLegacyEvent(element[i]);
 68           }
 69         } else {
 70           addLegacyEvent(element);
 71         }
 72       } else {
 73         element.addEventListener(eventType, eventFunction);
 74       }
 75 
 76       function addLegacyEvent(element) {
 77         var oldEvent = eval('element.on' + eventType);
 78         var eventContentType = eval('typeof element.on' + eventType);
 79 
 80         if (eventContentType != 'function') {
 81           eval('element.on' + eventType + ' = eventFunction');
 82         } else {
 83           eval('element.on' + eventType +
 84                ' = function(e) { oldEvent(e); eventFunction(e); }');
 85         }
 86       }
 87     }
 88   };
 89   /* End public */
 90 })();
