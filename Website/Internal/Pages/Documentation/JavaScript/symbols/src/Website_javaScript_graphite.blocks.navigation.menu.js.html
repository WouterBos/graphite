 15 /**
 16  * @namespace Menu functionality
 17  * @class
 18  * @since 1.0 - 2010-02-23
 19  * @version 1.0 - 2010-02-23

 21 graphite.blocks.navigation.menu = (function() {
 22   /* Start public */
 23   return {
 24     /**
 25      * Connects a function to an event of a single element
 26      *
 27      * @since 1.0 - 2011-11-03
 28      * @version 1.0 - 2011-11-03
 29      * @example
 30         graphite.events.addevent(
 31           document.getElementById('elementId'),
 32           function() {alert('foo')},
 33           "click",
 34           true
 35         );
 36      */
 37     initCollapse: function() {
 38       // Get the menu root element. We'll use jQuery for brevity
 39       var menuListItems = jQuery('.gp_menu.gp_menu_typeCollapse li');
 40       var cssDelay = new graphite.css.cssDelay(true);
 41 
 42       menuListItems.each(function() {
 43         cssDelay.addItem(
 44           {
 45             element: jQuery(this)[0],
 46             cssclass: 'gp_hover',
 47             eventType: 'mouseover',
 48             action: 'add',
 49             time: 0
 50           }
 51         );
 52         cssDelay.addItem(
 53           {
 54             element: jQuery(this)[0],
 55             cssclass: 'gp_hover',
 56             eventType: 'mousemove',
 57             action: 'add',
 58             time: 0
 59           }
 60         );
 61         cssDelay.addItem(
 62           {
 63             element: jQuery(this)[0],
 64             cssclass: 'gp_hover',
 65             eventType: 'mouseout',
 66             action: 'remove',
 67             time: 1000
 68           }
 69         );
 70       });
 71 
 72       cssDelay.createEvents();
 73     }
 74   };
 75   /* End public */
 76 })();

 87  */
 88 graphite.css = {};

 90 
 91

 94 /**
 95  * Adds and removes classes with a certain delay
 96  *
 97  * @since 1.0 - 2010-02-23
 98  * @version 1.0 - 2010-02-23
 99  * @constructor
100  *
101  * @param {Boolean} arg_singleSibling If true, only one sibling can have the
102  *    supplied class. Default value is false.
103  *
104  * @example
105    var cssDelay = new graphite.css.cssDelay(true);
106    cssDelay.addItem(
107      {
108        element: jQuery(this)[0],
109        cssclass: 'gp_hover',
110        eventType: 'mouseover',
111        action: 'add',
112        time: 0
113      }
114    );
115    cssDelay.addItem(
116      {
117        element: jQuery(this)[0],
118        cssclass: 'gp_hover',
119        eventType: 'mouseout',
120        action: 'remove',
121        time: 1000
122      }
123    );
124  */
125 graphite.css.cssDelay = function(arg_singleSibling) {
126   var _arr = new Array();
127   var _singleSibling = arg_singleSibling || false;
128 
129 
130 
131   /* Start public */
132   /**
133    * Adds and removes class with delay
134    *
135    * @param {Object} classEvent Configuration object with DOM element, class,
136    *    event, etc.
137    * @example
138      var cssDelay = new graphite.css.cssDelay(true);
139      cssDelay.addItem(
140        {
141          element: jQuery(this)[0],
142          cssclass: 'gp_hover',
143          eventType: 'mouseover',
144          action: 'add',
145          time: 0
146        }
147      );
148    */
149   this.addItem = function(classEvent) {
150     _arr.push(classEvent);
151   }
152 
153   this.createEvents = function() {
154     // Loop through array and create events.
155     for (var i = 0; i < _arr.length; i++) {
156       graphite.events.addEvent(
157         _arr[i].element,
158         (function(classEvent) {
159           return function() {
160               toggleClass(classEvent);
161           }
162         })(_arr[i]),
163         _arr[i].eventType,
164         true
165       );
166     }
167 
168     // The event handler.
169     function toggleClass(classEvent) {
170       var timeoutName = classEvent.action + '_' + classEvent.cssclass;
171 
172       // Create timeout object on element.
173       if (typeof(classEvent.element.gp_timeout) == 'undefined') {
174         classEvent.element.gp_timeout = {};
175       }
176 
177       // Clear possible existing timeouts when removing classes.
178       if (typeof(classEvent.element.gp_timeout[timeoutName]) != 'undefined') {
179         if (classEvent.action == 'remove') {
180           clearTimeout(classEvent.element.gp_timeout[timeoutName]);
181         }
182       }
183 
184       // The actual adding or removing of classes.
185       classEvent.element.gp_timeout[timeoutName] = setTimeout(
186         function() {
187           // Add class.
188           if (classEvent.action == 'add') {
189             var className = classEvent.element.className;
190             if (className.indexOf(classEvent.cssclass) == -1) {
191               classEvent.element.className += ' ' + classEvent.cssclass;
192             }
193             // Make sure only one sibling has that class.
194             if (_singleSibling == true && classEvent.eventType == 'mouseover') {
195               var parent = classEvent.element.parentNode;
196               for (var i = 0; i < parent.childNodes.length; i++) {
197                 if (parent.childNodes[i].nodeType == 1) {
198                   var classNameRegEx = new RegExp(classEvent.cssclass, 'g');
199                   var className = parent.childNodes[i].className;
200                   var newClassName = className.replace(classNameRegEx, '');
201                   parent.childNodes[i].className = newClassName;
202                 }
203               }
204             }
205           }
206           // Remove class.
207           if (classEvent.action == 'remove') {
208             var classNameRegEx = new RegExp(classEvent.cssclass, 'g');
209             var className = classEvent.element.className;
210             var newClassName = className.replace(classNameRegEx, '');
211             classEvent.element.className = newClassName;
212           }
213         },
214         classEvent.time
215       );
216     }
217   }
218   /* End public */
219 };
