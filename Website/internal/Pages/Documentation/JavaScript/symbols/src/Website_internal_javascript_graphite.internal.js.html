 20 /**
 21  * Extracts code from demo and presents to user.
 22  *
 23  * @since 1.0 - 2010-12-18
 24  * @version 1.0 - 2010-12-18
 25  *
 26  * @constructor
 27  * @param {Object} arg_config Config object.
 28  *    .root: The demo container.
 29  */
 30 graphite.internal.demo = function(arg_config) {
 31   var config = {
 32     root: arg_config.root
 33   };
 34 
 35   sourceCode = graphite.internal.demo.data.sourceCode;
 36 
 37   function setEvents() {
 38     var rootLinks = document.getElementsByTagName('a');
 39 
 40     for (var i = 0; i < rootLinks.length; i++) {
 41       // ASCX
 42       if (rootLinks[i].className.indexOf('graphite_getCodeAscx') != -1) {
 43         setEvent(
 44           rootLinks[i],
 45           sourceCode.ascx
 46         );
 47       }
 48       // Codebehind
 49       if (rootLinks[i].className.indexOf('graphite_getCodeCodeBehind') != -1) {
 50         setEvent(
 51           rootLinks[i],
 52           sourceCode.codebehind
 53         );
 54       }
 55       // HTML
 56       if (rootLinks[i].className.indexOf('graphite_getCodeHtml') != -1) {
 57         setEvent(
 58           rootLinks[i],
 59           sourceCode.html
 60         );
 61       }
 62       // CSS
 63       if (rootLinks[i].className.indexOf('graphite_getCodeCss') != -1) {
 64         setEvent(
 65           rootLinks[i],
 66           sourceCode.css
 67         );
 68       }
 69       // JavaScript
 70       if (rootLinks[i].className.indexOf('graphite_getCodeJs') != -1) {
 71         sourceCode.js = sourceCode.js.replace(/\#\#\#GP\#\#\#SCRIPT/g,
 72                                               '<script');
 73         sourceCode.js = sourceCode.js.replace(/\#\#\#GP\#\#\#\/SCRIPT/g,
 74                                               '</script');
 75         setEvent(
 76           rootLinks[i],
 77           sourceCode.js
 78         );
 79       }
 80     }
 81 
 82     function setEvent(copyLink, source) {
 83       if (copyLink && source) {
 84         var clipHtml = new ZeroClipboard.Client();
 85         clipHtml.setHandCursor(true);
 86 
 87         clipHtml.addEventListener('mouseup', function(client) {
 88           clipHtml.setText(source);
 89           var feedbackElement = jQuery('*.graphite_getCodeFeedback');
 90           feedbackElement.html('Code copied')
 91           feedbackElement.show();
 92           feedbackElement.css('opacity', 1);
 93           feedbackElement.delay(1500).animate(
 94             {
 95               opacity: 0
 96             },
 97             1000
 98           );
 99         });
100 
101         clipHtml.glue(copyLink);
102       }
103     }
104   }
105 
106   this.extractCode = function() {
107     // Setup cut 'n paste containers
108     graphite.events.addEvent(
109       window,
110       setEvents,
111       'load',
112       true
113     );
114   }
115 };
